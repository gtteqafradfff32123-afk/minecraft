private static void worldCollapseTick(ServerWorld clone, State st) {
Random r = clone.rand;
float progress = (float) st.ticks / st.durationTicks; // 0.0 тЖТ 1.0 ╨╖╨░ 6 ╨╝╨╕╨╜╤Г╤В
int events = 1 + (int)(progress * 8); // ╨║ ╨║╨╛╨╜╤Ж╤Г ╨┤╨╛ 9 ╨║╨░╤В╨░╤Б╤В╤А╨╛╤Д ╨╖╨░ ╤В╨╕╨║ ╤А╨░╤Б╨┐╨░╨┤╨░

int cx = st.center.getX() + (int)(Math.cos(ang) * rad);

} else if (type < 55) {

/** ╨Ю╨│╤А╨╛╨╝╨╜╨░╤П ╨┐╤А╨╛╨┐╨░╤Б╤В╤М: ╨▓╤Л╤А╨╡╨╖╨░╨╡╨╝ ╤Ж╨╕╨╗╨╕╨╜╨┤╤А ╨▓╨╜╨╕╨╖ ╨▓ ╨┐╤Г╤Б╤В╨╛╤В╤Г. */
private static void carveChasm(ServerWorld w, BlockPos c, Random r) {
int radius = 3 + r.nextInt(6); // 3-8 ╨▒╨╗╨╛╨║╨╛╨▓
int depth = 20 + r.nextInt(40); // ╨│╨╗╤Г╨▒╨╛╨║╨╛ ╨▓╨╜╨╕╨╖
for (int x = -radius; x <= radius; x++)
for (int z = -radius; z <= radius; z++) {
if (x*x + z*z > radius*radius) continue;
for (int y = 0; y < depth; y++) {
BlockPos bp = c.add(x, -y, z);
if (w.getBlockState(bp).getBlock() == Blocks.BARRIER) continue; // ╤Б╤В╨╡╨╜╤Г ╨╜╨╡ ╤В╤А╨╛╨│╨░╨╡╨╝
w.setBlockState(bp, Blocks.AIR.getDefaultState(), 18);
}
}
w.playSound(null, c, SoundEvents.ENTITY_GENERIC_EXPLODE, SoundCategory.BLOCKS, 2f, 0.3f);
}
/** ╨С╨╗╨╛╨║╨╕ ╨╛╤Б╤Л╨┐╨░╤О╤В╤Б╤П ╨▓╨╜╨╕╨╖ ╨║╨░╨║ ╨│╤А╨░╨▓╨╕╨╣ (╤А╨╡╨░╨╗╤М╨╜╤Л╨╡ ╨┐╨░╨┤╨░╤О╤Й╨╕╨╡ ╨▒╨╗╨╛╨║╨╕). */
private static void collapseColumn(ServerWorld w, BlockPos c, Random r) {
int radius = 2 + r.nextInt(4);
for (int x = -radius; x <= radius; x++)
for (int z = -radius; z <= radius; z++) {
BlockPos bp = c.add(x, 0, z);
BlockState bs = w.getBlockState(bp);
if (bs.isAir() || bs.getBlock() == Blocks.BARRIER || bs.getBlock() == Blocks.BEDROCK) continue;
// ╤Б╨┐╨░╨▓╨╜╨╕╨╝ ╤А╨╡╨░╨╗╤М╨╜╤Л╨╣ ╨┐╨░╨┤╨░╤О╤Й╨╕╨╣ ╨▒╨╗╨╛╨║ тЖТ ╨╛╨╜ ╨┐╨╛╨╗╨╡╤В╨╕╤В ╨▓ ╨┐╤Г╤Б╤В╨╛╤В╤Г
FallingBlockEntity fb = new FallingBlockEntity(w, bp.getX()+0.5, bp.getY(), bp.getZ()+0.5, bs);
fb.fallTime = 1;
w.addEntity(fb);
w.setBlockState(bp, Blocks.AIR.getDefaultState(), 18);
}
}
/** ╨Ф╤Л╤А╨░ ╨╜╨░╤Б╨║╨▓╨╛╨╖╤М, ╨▓╨║╨╗╤О╤З╨░╤П ╨▒╨╡╨┤╤А╨╛╨║ тАФ ╨┤╨╛ ╤Б╨░╨╝╨╛╨│╨╛ ╨╜╨╕╨╖╨░. */
private static void voidHole(ServerWorld w, BlockPos c, Random r) {

for (int x = -radius; x <= radius; x++)
for (int z = -radius; z <= radius; z++) {

for (int y = c.getY(); y >= 0; y--) {
BlockPos bp = new BlockPos(c.getX()+x, y, c.getZ()+z);
if (w.getBlockState(bp).getBlock() == Blocks.BARRIER) continue;
w.setBlockState(bp, Blocks.AIR.getDefaultState(), 18); // ╨┤╨░╨╢╨╡ ╨▒╨╡╨┤╤А╨╛╨║ ╨╕╤Б╤З╨╡╨╖╨░╨╡╤В
}
}

public static class State {

public BlockPos center, realReturnPos;
public int ticks = 0;

public static void tick(ServerWorld clone) {
for (State st : STATES.values()) {
if (!st.cloneReady) continue;
ServerPlayerEntity vp = clone.getServer().getPlayerList().getPlayerByUUID(st.victim);
if (vp == null) continue;
st.ticks++;
float progress = (float) st.ticks / st.durationTicks;
int minutesLeft = (st.durationTicks - st.ticks) / (60 * 20);
// ╨╕╤Б╤В╨╛╤Й╨╡╨╜╨╕╨╡
if (st.ticks % 60 == 0 && vp.getHealth() > 1f)
vp.attackEntityFrom(DamageSource.WITHER, 1f);
// === ╨Ъ╨Ю╨Я╨Ш╨п: ╤Б╨┐╨░╨▓╨╜ 100% ╨╡╤Б╨╗╨╕ ╨╜╨╡╤В ╨░╨║╤В╨╕╨▓╨╜╨╛╨╣ ===
// ╤З╨╡╨╝ ╨▒╨╗╨╕╨╢╨╡ ╨║ ╤Б╨╝╨╡╤А╤В╨╕ (╨▒╨╛╨╗╤М╤И╨╡ copyKills), ╤В╨╡╨╝ ╨а╨Х╨Ц╨Х ╨┐╨╛╤П╨▓╨╗╤П╨╡╤В╤Б╤П

if (st.copiesAlive == 0 && st.copyKills < 3 && st.ticks % spawnDelay == 0) {
spawnPlayerCopy(clone, st, vp);
// ╨╕╨╜╨╛╨│╨┤╨░ ╨Ф╨Т╨Х ╤Б╤А╨░╨╖╤Г (╨┐╨╛╤Б╨╗╨╡ ╨┐╨╡╤А╨▓╨╛╨╣ ╤Б╨╝╨╡╤А╤В╨╕)

// === ╤А╨░╨╖╤А╤Г╤И╨╡╨╜╨╕╨╡ ╨╝╨╕╤А╨░ (╤Г╤Б╨╕╨╗╨╕╨▓╨░╨╡╤В╤Б╤П) ===
if (st.ticks % 20 == 0) worldCollapseTick(clone, st);
// === ╨н╨б╨Ъ╨Р╨Ы╨Р╨ж╨Ш╨п ╨▓╨╛ ╨▓╤В╨╛╤А╨╛╨╣ ╨┐╨╛╨╗╨╛╨▓╨╕╨╜╨╡: ghost-╨╝╨╛╨▒╤Л, ╨║╨╗╨╛╨╜╤Л ╤Б ╨░╨╗╨╝╨░╨╖╨║╨╛╨╣ ===
